    \bigskip

    \bigskip
    \textbf{RE: \paperid}

%% RESPONSE COMMANDS %%

% Need indented paragraph environment
\newenvironment{myindentpar}[1]%
  {\begin{list}{}%
          {\setlength{\leftmargin}{#1}}%
          \item[]%
  }
  {\end{list}}

% response allows for you to respond to reviewers comments, like this:
%       response[Line X]{Reviewer's Comment}{Your Reply}[Quote][Quote2]
% all of the options are, naturally, optional. The only required field is the reviewer's response.

\makeatletter
\DeclareDocumentCommand{\response}{ o m m o o }{%
        \IfNoValueTF{#1}%
            {}%
            {\@ifmtarg{#1}{\textit{#2}}{#1: \textit{#2}}}%
        \begin{myindentpar}{1cm}#3\end{myindentpar}%
        \IfNoValueTF{#4}%
            {}%
            {\@ifmtarg{#4}{}{\begin{myindentpar}{1cm}\textcolor{\quotecolor}{``#4\xspace''}\end{myindentpar}}}%
        \IfNoValueTF{#5}%
            {}%
            {\@ifmtarg{#5}{}{\begin{myindentpar}{1cm}\textcolor{\quotecolor}{``#5\xspace''}\end{myindentpar}}}%
}
\makeatother
